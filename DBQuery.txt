CREATE TABLE `misc` (
 `misc_id` int(11) NOT NULL,
 `Name` text NOT NULL,
 `About` text NOT NULL,
 `Image` longblob NOT NULL,
 PRIMARY KEY (`misc_id`)
);

CREATE TABLE `showcase` (
 `showcase_id` int(11) NOT NULL,
 `Title` text NOT NULL,
 `Description` text NOT NULL,
 `Image_Full` longblob NOT NULL,
 `Image_Thumbnail` longblob NOT NULL,
 `Image_Mobile` longblob NOT NULL,
 `order` int(11) NOT NULL,
 PRIMARY KEY (`showcase_id`)
);

CREATE TABLE `skill` (
 `skill_id` int(11) NOT NULL,
 `Title` text NOT NULL,
 `Description` text NOT NULL,
 PRIMARY KEY (`skill_id`)
);

CREATE TABLE `showcase_skill` (
 `showcase_skill_id` int(11) NOT NULL,
 `showcase_id` int(11) NOT NULL,
 `skill_id` int(11) NOT NULL,
 PRIMARY KEY (`showcase_skill_id`)
);
